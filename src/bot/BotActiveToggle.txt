import { useState } from "react"
import { Switch } from "@/components/ui/switch"
import { Loader, Power } from "lucide-react"
import { useAppKitAccount } from "@reown/appkit/react"
import api from '@/utils/api'
import useToast from "@/components/Toast"

export default function BotActiveToggle({ botActive, setBotActive, userBotData, loadingBotData, onStatusChange }) {
    const { address, isConnected } = useAppKitAccount()
    const { toast } = useToast()
    const [loading, setLoading] = useState(false)

    const handleToggle = async (checked) => {
        if (!isConnected || !userBotData.subscribed) return

        setLoading(true)

        try {
            const endpoint = checked ? '/dca-bot/start' : '/dca-bot/stop'
            const { data } = await api.post(endpoint)

            setBotActive(checked)
            onStatusChange?.(checked ? 'active' : 'inactive')
            toast({
                title: `Bot ${checked ? 'started' : 'stopped'}`,
                description: `Trading bot is now ${checked ? 'active' : 'inactive'}`,
                variant: 'success',
            })
        } catch (error) {
            console.error('Bot toggle error:', error)
            const errorMessage = error?.response?.data?.message || 'Please try again'
            toast({
                title: 'Failed to update bot status',
                description: errorMessage,
                variant: 'error',
            })
        } finally {
            setLoading(false)
        }
    }

    return (
        <div className={`rounded-lg p-4  ${!userBotData.subscribed || !botActive
            ? 'bg-amber-500/10'
            : 'bg-slate-800/30'
            }`}>
            <div className="flex items-center justify-between">
                <div className="flex gap-2 items-center">
                    <Power className={`h-5 w-5 ${!userBotData.subscribed || !botActive
                        ? 'text-amber-400'
                        : 'text-white'
                        }`} />
                    <span className={`font-medium ${!userBotData.subscribed || !botActive
                        ? 'text-amber-300'
                        : 'text-white'
                        }`}>Bot Active</span>
                </div>
                {loadingBotData ? (
                    <Loader className="h-5 w-5 animate-spin text-slate-400" />
                ) : (
                    <button
                        onClick={() => handleToggle(!botActive)}
                        disabled={!userBotData.subscribed || loading}
                        className={`relative inline-flex h-6 w-14 items-center rounded-full transition-colors duration-200 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-offset-slate-800 ${botActive
                                ? 'bg-green-600 focus:ring-green-500'
                                : 'bg-red-600 focus:ring-red-500'
                            } ${(!userBotData.subscribed || loading) ? 'opacity-50 cursor-not-allowed' : 'cursor-pointer'}`}
                    >
                        <span className={`inline-block h-4 w-4 transform rounded-full bg-white/80 transition-transform duration-200 ${botActive ? 'translate-x-9' : 'translate-x-1'
                            }`} />
                        <span className={`absolute inset-0 flex items-center text-xs font-medium text-white ${botActive ? 'justify-start pl-2' : 'justify-end pr-2'
                            }`}>
                            {botActive ? 'ON' : 'OFF'}
                        </span>
                    </button>
                )}
            </div>
        </div>
    )
}
